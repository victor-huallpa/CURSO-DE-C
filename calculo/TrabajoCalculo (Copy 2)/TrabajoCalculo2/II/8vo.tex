Hallar el área de la región acotada por $y=5\sqrt{x}$, el eje $x$ y las rectas $x=1$ y $x=4$.

\nopagebreak
\begin{align*}
\text{Intervalo: } [1,4]. \quad \Delta x = \frac{3}{n}. \quad x_i = 1 + \frac{3i}{n}. \\
\text{Suma de Riemann: } A = \lim_{n \to \infty} \sum_{i=1}^{n} 5\sqrt{1+\frac{3i}{n}} \cdot \frac{3}{n} \\
\text{Evaluando la integral:} \\
A = \int_{1}^{4} 5\sqrt{x} \, dx = 5 \int_{1}^{4} x^{1/2} \, dx \\
= 5 \left[ \frac{2}{3} x^{3/2} \right]_{1}^{4} = \frac{10}{3} (4^{3/2} - 1^{3/2}) \\
= \frac{10}{3} (8 - 1) = \frac{70}{3}
\end{align*}

\[
\boxed{\displaystyle \frac{70}{3}}
\]
