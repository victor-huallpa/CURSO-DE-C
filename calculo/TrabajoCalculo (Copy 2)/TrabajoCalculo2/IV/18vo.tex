$x=1+y^2$, $x=10$

\nopagebreak
\begin{align*}
\text{Intersección: } 1+y^2 = 10 \implies y^2=9 \implies y=\pm 3. \\
\text{Integramos respecto a } y: \\
A &= \int_{-3}^{3} (10 - (1+y^2)) \, dy = \int_{-3}^{3} (9-y^2) \, dy \\
&= \left[ 9y - \frac{y^3}{3} \right]_{-3}^{3} \\
&= (27 - 9) - (-27 - (-9)) \\
&= 18 - (-18) = 36
\end{align*}

\[
\boxed{\displaystyle 36}
