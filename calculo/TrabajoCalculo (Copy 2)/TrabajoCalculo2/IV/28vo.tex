\begin{align*}
\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1 \implies y = \pm b\sqrt{1 - x^2/a^2}. \\
A = 4 \int_{0}^{a} b\sqrt{1 - \frac{x^2}{a^2}} \, dx \\
\text{Sustitución: } x = a\sin\theta, dx = a\cos\theta d\theta. \\
A = 4b \int_{0}^{\pi/2} \sqrt{1-\sin^2\theta} \cdot a\cos\theta \, d\theta \\
= 4ab \int_{0}^{\pi/2} \cos^2\theta \, d\theta \\
= 4ab \cdot \frac{1}{2} \cdot \frac{\pi}{2} = \pi ab
\end{align*}
