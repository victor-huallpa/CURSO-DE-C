\nopagebreak
\begin{align*}
\text{Intersección: } & x^4-4x^2 = 4x^2 \implies x^4-8x^2=0 \\
& \implies x=0, x=\pm\sqrt{8}. \\
A &= 2 \int_{0}^{\sqrt{8}} (8x^2 - x^4) \, dx \\
&= 2 \left[ \frac{8x^3}{3} - \frac{x^5}{5} \right]_{0}^{\sqrt{8}} \\
&= 2 \left( \frac{8(8\sqrt{8})}{3} - \frac{64\sqrt{8}}{5} \right) \\
&= 128\sqrt{8} \left( \frac{1}{3} - \frac{1}{5} \right) \\
&= 128(2\sqrt{2}) \frac{2}{15} = \frac{512\sqrt{2}}{15}
\end{align*}
