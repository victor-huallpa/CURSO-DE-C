$y^2=x^3$, $y=0$, $x=2$; eje $x$.

\nopagebreak
\begin{align*}
y = x^{3/2}. \text{ Rotación alrededor del eje } x. \\
\text{Método de discos:} \\
V &= \pi \int_{0}^{2} (x^{3/2})^2 \, dx = \pi \int_{0}^{2} x^3 \, dx \\
&= \pi \left[ \frac{x^4}{4} \right]_{0}^{2} \\
&= \pi \left( \frac{16}{4} - 0 \right) = 4\pi
\end{align*}

\[
\boxed{\displaystyle 4\pi}
